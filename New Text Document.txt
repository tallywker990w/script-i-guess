local Rayfield = loadstring(game:HttpGet('https://sirius.menu/rayfield'))()

local Window = Rayfield:CreateWindow({
   Name = "Prism Hub | Bee Swarm Simulator 🐝",
   LoadingTitle = "Loading Prism Scripts...",
   LoadingSubtitle = "by tallywker990",
   ConfigurationSaving = { Enabled = false },
   Discord = {
      Enabled = true,
      Invite = "2BSmXd8hKY",
      RememberJoins = true
   }
})

local Main = Window:CreateTab("Auto-Farm", 4483362458)

-- Variables
local lp = game.Players.LocalPlayer
local selectedField = "Sunflower Field"

-- RELIABLE HIVE FINDER
local function getMyHive()
    for _, hive in pairs(workspace.Hives:GetChildren()) do
        if hive:FindFirstChild("Owner") and hive.Owner.Value == lp.Name then
            return hive.Floor -- Return the part we can TP to
        end
    end
    -- Fallback: Look for the slot with your name on the board
    for _, slot in pairs(workspace.Hives:GetChildren()) do
        local surface = slot:FindFirstChild("SurfaceGui")
        if surface and surface:FindFirstChild("PlayerName") and surface.PlayerName.Text == lp.Name then
            return slot.Floor
        end
    end
    return nil
end

-- STAT FINDER
local function getStats()
    local s = lp:FindFirstChild("CoreStats") or lp:FindFirstChild("Data") or lp:FindFirstChild("leaderstats")
    if s then
        local p = s:FindFirstChild("Pollen")
        local c = s:FindFirstChild("Capacity")
        if p and c then return p, c end
    end
    return nil, nil
end

Main:CreateDropdown({
   Name = "Target Field",
   Options = {"Sunflower Field", "Mushroom Field", "Clover Field", "Blue Flower Field", "Dandelion Field", "Strawberry Field", "Spider Field", "Bamboo Field", "Pineapple Patch", "Pine Tree Forest"},
   CurrentOption = "Sunflower Field",
   Callback = function(Option) selectedField = Option end,
})

Main:CreateToggle({
   Name = "Enable Full Auto-Loop",
   CurrentValue = false,
   Callback = function(Value)
       _G.AutoLoop = Value
       task.spawn(function()
           while _G.AutoLoop do
               local pStat, cStat = getStats()
               
               if pStat and cStat then
                   -- If Bag is Full (95% to be safe)
                   if pStat.Value >= (cStat.Value * 0.95) then
                       Rayfield:Notify({Title = "Farm", Content = "Bag Full! Going to Hive."})
                       
                       local myHive = getMyHive()
                       if myHive then
                           -- Loop until empty
                           repeat
                               lp.Character.HumanoidRootPart.CFrame = myHive.CFrame + Vector3.new(0, 5, 0)
                               task.wait(1)
                           until pStat.Value == 0 or not _G.AutoLoop
                           Rayfield:Notify({Title = "Farm", Content = "Bag Empty! Returning to Field."})
                       else
                           Rayfield:Notify({Title = "Error", Content = "Could not find your Hive!"})
                       end
                   else
                       -- Farm Logic
                       local field = workspace.FlowerZones:FindFirstChild(selectedField)
                       if field then
                           -- TP to field if far away
                           if (lp.Character.HumanoidRootPart.Position - field.Position).Magnitude > 30 then
                               lp.Character.HumanoidRootPart.CFrame = field.CFrame + Vector3.new(0, 5, 0)
                           end
                           
                           -- Click/Swing
                           local tool = lp.Character:FindFirstChildOfClass("Tool")
                           if tool then tool:Activate() end
                           
                           -- Grab nearby tokens
                           for _, v in pairs(workspace.Collectibles:GetChildren()) do
                               if v:IsA("BasePart") and (v.Position - lp.Character.HumanoidRootPart.Position).Magnitude < 25 then
                                   v.CFrame = lp.Character.HumanoidRootPart.CFrame
                               end
                           end
                       end
                   end
               end
               task.wait(0.1)
           end
       end)
   end,
})
